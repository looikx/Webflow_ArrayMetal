  function init() {
    var root = document.getElementById('oa-top');
    if (!root) return;

    // Build hero
    var hero = document.createElement('div');
    hero.className = 'oa-hero';
    hero.innerHTML = '<div class="oa-hero-label">Array Metal (M) Sdn. Bhd.</div>' +
      '<h1>Open Area Calculator</h1>' +
      '<p>Select a perforation pattern to calculate open area percentage and hole density for your perforated metal sheet.</p>';
    root.appendChild(hero);

    // Build thumb section
    var thumbSection = document.createElement('div');
    thumbSection.id = 'oa-thumb-section';
    root.appendChild(thumbSection);

    // Build sections wrap
    var sectionsWrap = document.createElement('div');
    sectionsWrap.className = 'oa-sections-wrap';
    sectionsWrap.id = 'oa-sections';
    root.appendChild(sectionsWrap);

    buildThumbs(thumbSection);
    buildSections(sectionsWrap);
  }

      hdr.textContent = g.label;
      wrap.appendChild(hdr);
      var grid = document.createElement('div');
      grid.className = 'oa-thumb-grid';
      g.keys.forEach(function(key) {
        var p = PATTERNS[key];
        var card = document.createElement('a');
        card.className = 'oa-thumb-card';
        card.href = '#';
        card.setAttribute('data-target', 'oa-section-' + key);
        var img = document.createElement('img');
        img.src = IMGS[key];
        img.alt = p.name;
        card.appendChild(img);
        var name = document.createElement('div');
        name.className = 'oa-thumb-name';
        name.textContent = p.name;
        card.appendChild(name);
        var desc = document.createElement('div');
        desc.className = 'oa-thumb-desc';
        desc.textContent = p.desc;
        card.appendChild(desc);
        card.addEventListener('click', function(e) {
          e.preventDefault();
          var sec = document.getElementById(this.getAttribute('data-target'));
          if (!sec) return;
          var navEl = document.querySelector('.navbar.w-nav') || document.querySelector('.w-nav');
          var navH = navEl ? navEl.offsetHeight : 81;
          var top = sec.getBoundingClientRect().top + window.pageYOffset - navH - 16;
          window.scrollTo({ top: top, behavior: 'smooth' });
        });
        grid.appendChild(card);
      });
      wrap.appendChild(grid);
    });
  }

    GROUPS.forEach(function(g) { g.keys.forEach(function(k) { keys.push(k); }); });
    keys.forEach(function(key, idx) {

      if (idx > 0) { var hr = document.createElement('hr'); hr.className = 'oa-section-divider'; wrap.appendChild(hr); }
      var sec = document.createElement('div');
      sec.className = 'oa-pattern-section';
      sec.id = 'oa-section-' + key;
      var hdr = document.createElement('div');
      hdr.className = 'oa-section-heading';

      });
    });

  function calculate(key) {
    var p = PATTERNS[key];
    var v = {}, allFilled = true;
    p.fields.forEach(function(f) {
      var el = document.getElementById('oa-' + key + '-' + f.id);
      var val = parseFloat(el.value);
      if (!el.value || isNaN(val) || val <= 0) allFilled = false;
      v[f.id] = val;
    });
    var oaEl = document.getElementById('oa-oa-' + key);
    var holesEl = document.getElementById('oa-holes-' + key);
    if (!allFilled || !p.valid(v)) {
      oaEl.textContent = '—'; oaEl.className = 'oa-result-value';
      holesEl.textContent = '—'; return;
    }
    var oa = p.openArea(v), holes = p.holesPerM2(v);
    if (oa < 0 || oa > 100) { oaEl.textContent = '—'; oaEl.className = 'oa-result-value'; holesEl.textContent = '—'; return; }
    oaEl.innerHTML = oa.toFixed(1) + '<span class="oa-result-unit">%</span>';
    oaEl.className = 'oa-result-value ' + (oa < 20 ? 'oa-low' : oa < 40 ? 'oa-medium' : oa < 60 ? 'oa-good' : 'oa-high');
    holesEl.innerHTML = Math.round(holes).toLocaleString() + '<span class="oa-holes-unit">/m²</span>';
  }

  function resetCalc(key) {
    var p = PATTERNS[key];
    p.fields.forEach(function(f) {
      var el = document.getElementById('oa-' + key + '-' + f.id);
      if (el) el.value = '';
    });
    var oaEl = document.getElementById('oa-oa-' + key);
    if (oaEl) { oaEl.textContent = '—'; oaEl.className = 'oa-result-value'; }
    var holesEl = document.getElementById('oa-holes-' + key);
    if (holesEl) holesEl.textContent = '—';
  }

  function sendInquiry(key, btn, msgEl) {
    var p = PATTERNS[key];
    var lines = [];
    p.fields.forEach(function(f) {
      var el = document.getElementById('oa-' + key + '-' + f.id);
      var val = el ? el.value.trim() : '';
      if (val) lines.push(f.label + ': ' + val + ' mm');
    });
    var oaEl = document.getElementById('oa-oa-' + key);
    var oaText = oaEl ? oaEl.textContent.replace(/<[^>]+>/g, '').trim() : '—';
    var holesEl = document.getElementById('oa-holes-' + key);
    var holesText = holesEl ? holesEl.textContent.replace(/<[^>]+>/g, '').trim() : '—';

    btn.disabled = true;
    msgEl.textContent = 'Sending…';
    msgEl.className = 'oa-inquiry-msg';

    var payload = {
      name: 'Open Area Inquiry',
      source: window.location.href,
      'test': false,
      dolphin: false,
      fields: {
        'Pattern': p.name + ' \u2014 ' + p.desc,
        'Dimensions': lines.length ? lines.join(', ') : '(not entered)',
        'Open Area': oaText,
        'Holes per m\u00b2': holesText
      }
    };

    fetch('https://webflow.com/api/v1/form/6082b34dc5995b3e8dc8c73b', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    }).then(function(r) {
      btn.disabled = false;
      if (r.ok) {
        msgEl.textContent = 'Inquiry sent! We\u2019ll be in touch.';
        msgEl.className = 'oa-inquiry-msg oa-ok';
      } else {
        msgEl.textContent = 'Submission failed (error ' + r.status + '). Please contact us directly.';
        msgEl.className = 'oa-inquiry-msg oa-err';
      }
    }).catch(function() {
      btn.disabled = false;
      msgEl.textContent = 'Could not send. Please contact us directly.';
      msgEl.className = 'oa-inquiry-msg oa-err';
    });
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
